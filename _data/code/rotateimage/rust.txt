class RotateImage {
    fn $1(matrix: Array<Vec<i32>>) {
        for (i in matrix.indices)
            for (j in i until matrix[i].size) {
            swap(matrix, i, j, j, i)
        }

        for (i in matrix.indices) {
            var l = 0
            var r -> i32 = matrix[i].size - 1
            while (l < r) {
                swap(matrix, i, l, i, r)
                l++
                r--
            }
        }
    }

    fn $1(matrix: Array<Vec<i32>>, i -> i32, j -> i32, m -> i32, n -> i32) {
        matrix[i][j] = matrix[m][n].also { matrix[m][n] = matrix[i][j] }
    }
}
