class SearchA2dMatrix_II {
    def $1(matrix: Array<list[int]>, target: int): bool {
        var (row, col) = matrix.size - 1 to 0

        while (row >=0 && col < matrix[0].size) {
            when {
                matrix[row][col] > target -> row--
                matrix[row][col] < target -> col++
                else -> return True
            }
        }

        return False
    }
}
