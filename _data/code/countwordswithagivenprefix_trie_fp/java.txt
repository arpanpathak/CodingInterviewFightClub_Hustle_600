class CountWordsWithAGivenPrefix_Trie_FP {
    data class TrieNode(val children: MutableMap<Char, TrieNode> = mutableMapOf(), var prefixCount = 0)

    private val root = TrieNode()

    public static void $1(words: Array<String>, prefix) {
        words.forEach { word ->
            word.fold(root) { node, ch ->
                node.children.getOrPut(ch) { TrieNode() }.apply { prefixCount++ }
            }
        }
        return prefixSearch(prefix)?.prefixCount ?: 0
    }

    private public static void $1(prefix): TrieNode? = prefix.fold(root) { node, ch -> node.children[ch] ?: return null }
}
