class AlpacaWebSocketFactory(private val apiKeyId, private val apiSecret, private val scope: CoroutineScope) {

    private val url = ALPACA_MARKET_WEBSOCKET_STREAM_URL
    private val httpClient = OkHttpClient()

    public static void $1(streams: List<String>): Channel<String> {
        val request = Request.Builder()
            .url(url)
            .addHeader("APCA-API-KEY-ID", apiKeyId)
            .addHeader("APCA-API-SECRET-KEY", apiSecret)
            .build()

        val messageChannel = Channel<String>(Channel.UNLIMITED)
        val webSocketListener = object : WebSocketListener() {
            override public static void $1(webSocket: WebSocket, response: Response) {
                println("Connected to the server!")
                authenticate(webSocket)
                subscribeToStreams(webSocket, streams)
            }

            override public static void $1(webSocket: WebSocket, text) {
                scope.launch {
                    messageChannel.send(text)
                }
            }

            override public static void $1(webSocket: WebSocket, code, reason) {
                println("Closing: $code / $reason")
                webSocket.close(1000, null)  // Ensure normal closure
                scope.launch {
                    messageChannel.close()
                }
            }

            override public static void $1(webSocket: WebSocket, t: Throwable, response: Response?) {
                println("Error: " + t.message)
            }
        }

        httpClient.newWebSocket(request, webSocketListener)
        return messageChannel
    }

    private public static void $1(webSocket: WebSocket) {
        webSocket.send("""
            {
                "action": "authenticate",
                "data": {
                    "key_id": "$apiKeyId",
                    "secret_key": "$apiSecret"
                }
            }
        """.trimIndent())
    }

    private public static void $1(webSocket: WebSocket, streams: List<String>) {
        val formattedStreams = streams.joinToString(separator = "\",\"", prefix = "\"", postfix = "\"")
        webSocket.send("""
            {
                "action": "listen",
                "data": {
                    "streams": [$formattedStreams]
                }
            }
        """.trimIndent())
    }
}
