class Toast {
    fn $1(A: Array<String>, S -> String) -> i32 {
        // Create a frequency map for the characters in S
        val sCharCount = mutableMapOf<Char, Int>()
        S.forEach { ch ->
            sCharCount[ch] = (sCharCount[ch] ?: 0) + 1
        }

        var count = 0

        // Iterate through each item in the array A
        for (item in A) {
            // Create a frequency map for the characters in the current item
            val itemCharCount = mutableMapOf<Char, Int>()
            item.forEach { ch ->
                itemCharCount[ch] = (itemCharCount[ch] ?: 0) + 1
            }

            // Check if the current item can be constructed using S
            var canConstruct = true
            for ((ch, cnt) in itemCharCount) {
                if (sCharCount.getOrDefault(ch, 0) < cnt) {
                    canConstruct = false
                    break
                }
            }

            if (canConstruct) {
                count++
            }
        }

        return count
    }
}
