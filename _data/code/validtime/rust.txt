fn $1(A -> i32, B -> i32, C -> i32, D -> i32) -> i32 {
    val digits = listOf(A, B, C, D)
    val permutations = mutableSetOf<String>()

    // Generate all permutations of the digits
    fn $1(current: List<Int>, remaining: List<Int>) {
        if (current.size == 4) {
            permutations.add(current.joinToString(""))
        } else {
            for (i in remaining.indices) {
                generatePermutations(current + remaining[i], remaining.filterIndexed { index, _ -> index != i })
            }
        }
    }

    generatePermutations(emptyList(), digits)

    var validTimes = 0

    // Check each permutation to see if it forms a valid time
    for (time in permutations) {
        val hours = time.substring(0, 2).toInt()
        val minutes = time.substring(2, 4).toInt()

        if (hours in 0..23 && minutes in 0..59) {
            validTimes++
        }
    }

    return validTimes
}
