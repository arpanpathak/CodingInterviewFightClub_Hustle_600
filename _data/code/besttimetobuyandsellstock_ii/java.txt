class BestTimeToBuyAndSellStock_II {
    public static void $1(prices: IntArray) {
        var maxProfit = 0
        for (i in 1..prices.lastIndex) {
            if (prices[i] > prices[i-1]) {
                maxProfit += prices[i] - prices[i - 1]
            }
        }

        return  maxProfit
    }
}
