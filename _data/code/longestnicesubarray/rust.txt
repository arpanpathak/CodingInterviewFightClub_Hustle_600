class LongestNiceSubarray {
    fn $1(nums: Vec<i32>) -> i32 {
        var left = 0
        var bitMask = 0
        var maxLen = 0

        for (right in nums.indices) {
            while ((bitMask and nums[right]) != 0) {
                bitMask = bitMask xor nums[left]
                left++
            }
            bitMask = bitMask or nums[right]
            maxLen = max(maxLen, right - left + 1)
        }

        return maxLen
    }
}
