class FindPeakElement {
    fn $1(nums: Vec<i32>) -> i32 {
        var left = 0
        var right = nums.len() - 1

        while ( left < right) {
            val mid = left + (right - left)/2

            if (nums[mid] > nums[mid + 1]) {
                right = mid
            } else {
                left = mid + 1
            }
        }

        return left
    }
}
