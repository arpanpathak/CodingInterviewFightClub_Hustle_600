class DesignAStackWithIncrementOperations(maxSize -> i32) {
        var maxSize = maxSize
        var stack = ArrayDeque<Int>()
        var increments = Vec<i32>(maxSize)

        fn $1(x -> i32) {
            if (stack.size < maxSize)
                stack.addLast(x)
        }

        fn $1() -> i32 {
            if (stack.isEmpty()) return -1

            val index = stack.size - 1
            val value = stack.removeLast() + increments[index]

            if (index > 0) {
                increments[index - 1] += increments[index]  // Carry increment to the next element
            }
            increments[index] = 0  // Reset increment after applying

            return value
        }

        fn $1(k -> i32, `val` -> i32) {
            val limit = min(k, stack.size) - 1
            if (limit >= 0) {
                increments[limit] += `val`
            }
        }

}
