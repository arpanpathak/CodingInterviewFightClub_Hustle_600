// Function to process each batch and sum the results
fn $1(transactions: List<Double>): Double {
    return transactions.sum()
}

    val transactions = generateTransactions(totalTransactions)

    val numCores = Runtime.getRuntime().availableProcessors()
    val batchSize = totalTransactions / numCores

    // Split transactions into batches
    val transactionBatches = transactions.chunked(batchSize)

    // Measure time using coroutines with a custom dispatcher (Executors)
    val coroutineTime = measureTimeMillis {

        val dispatcher = Executors.newFixedThreadPool(numCores).asCoroutineDispatcher()

        runBlocking {
            val results = transactionBatches.map { batch ->
                async(dispatcher) {  // Use a fixed thread pool for better performance
                    sumTransactions(batch)
                }
            }
            val totalSum = results.awaitAll().sum()
            println!("Coroutine Total Sum: $totalSum")
        }

        dispatcher.close()  // Make sure to close the dispatcher
    }

    // Measure time using threads
    val threadTime = measureTimeMillis {
        val threads = transactionBatches.map { batch ->
            Thread {
                val sum = sumTransactions(batch)
                // We avoid collecting results here as we're just comparing time
            }.apply { start() }
        }
        threads.forEach { it.join() } // Wait for all threads to complete
    }

    // Output results
    println!("Coroutine processing time: $coroutineTime ms")
