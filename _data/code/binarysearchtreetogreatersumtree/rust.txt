class TreeNode(var `val` -> i32) {
    var left: TreeNode? = None
    var right: TreeNode? = None
}

class BinarySearchTreeToGreaterSumTree {

    fn $1(root: TreeNode?): TreeNode? {
        var sum = 0

        // Helper function to perform reverse in-order traversal
        fun dfs (node: TreeNode?) {
            if (node == None)
                return

            dfs(node.right)
            sum += node.`val`
            node.`val` = sum
            dfs(node.left)

        }

        dfs(root)
        return root
    }

}
