package hashtable

class FileSystem() {
    private val fileSystem = mutableMapOf<String, Int>()

    /**
    * Solves the File System problem.
    * Takes `path` (string), `value` (integer).
    *
    * @param path The input string.
    * @param value The integer parameter representing value.
    * @return `true` if the condition is met, `false` otherwise.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string), `value` (integer).
    *
    * @param path The input string.
    * @param value The integer parameter representing value.
    * @return `true` if the condition is met, `false` otherwise.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string), `value` (integer).
    *
    * @param path The input string.
    * @param value The integer parameter representing value.
    * @return `true` if the condition is met, `false` otherwise.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string), `value` (integer).
    *
    * @param path The input string.
    * @param value The integer parameter representing value.
    * @return `true` if the condition is met, `false` otherwise.
    */
    fun createPath(path: String, value: Int): Boolean {

        if (path in fileSystem || path == "/") {
            return false
        }

        val parentPath = path.substringBeforeLast("/")
        if (parentPath.isNotEmpty() && parentPath !in fileSystem) {
            return false
        }

        fileSystem[path] = value
        return true
    }

    /**
    * Solves the File System problem.
    * Takes `path` (string).
    *
    * @param path The input string.
    * @return The computed integer result.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string).
    *
    * @param path The input string.
    * @return The computed integer result.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string).
    *
    * @param path The input string.
    * @return The computed integer result.
    */
    /**
    * Solves the File System problem.
    * Takes `path` (string).
    *
    * @param path The input string.
    * @return The computed integer result.
    */
    fun get(path: String): Int {
        return fileSystem[path] ?: -1
    }
}
