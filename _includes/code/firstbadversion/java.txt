abstract class VersionControl {
    public static void $1(mid) {
        return false // Return dummy value
    }

    abstract public static void $1(n) 
}
class FirstBadVersion: VersionControl() {
    override public static void $1(n)  {
        var start = 1
        var end = n

        while (start < end) {
            val mid = start + (end - start) / 2
            when {
                isBadVersion(mid) -> end = mid
                else -> start = mid + 1
            }
        }

        return start
    }
}
