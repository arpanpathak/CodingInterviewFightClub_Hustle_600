package quicksort

import kotlin.random.Random

class KthLargestElementInArrayTailRec {
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `left` (integer), `right` (integer).
    *
    * @param nums The input array of integers.
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `left` (integer), `right` (integer).
    *
    * @param nums The input array of integers.
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `left` (integer), `right` (integer).
    *
    * @param nums The input array of integers.
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `left` (integer), `right` (integer).
    *
    * @param nums The input array of integers.
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    fun partition(nums: IntArray, left: Int, right: Int): Int {
        // Randomly select pivot index and swap with the last element
        val pivotIndex = Random.nextInt(left, right + 1)
        nums[pivotIndex] = nums[right].also { nums[right] = nums[pivotIndex] }
        val pivot = nums[right]

        // Partition around the pivot
        var partitionIndex = left
        for (i in left until right) {
            if (nums[i] >= pivot)
                swap(nums, i, partitionIndex++)
        }

        swap(nums, partitionIndex, right)
        return partitionIndex
    }

    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `k` (integer).
    *
    * @param nums The input array of integers.
    * @param k The integer parameter representing k.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `k` (integer).
    *
    * @param nums The input array of integers.
    * @param k The integer parameter representing k.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `k` (integer).
    *
    * @param nums The input array of integers.
    * @param k The integer parameter representing k.
    * @return The computed integer result.
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `k` (integer).
    *
    * @param nums The input array of integers.
    * @param k The integer parameter representing k.
    * @return The computed integer result.
    */
    fun findKthLargest(nums: IntArray, k: Int): Int {
        var left = 0
        var right = nums.size - 1

        while (left <= right) {
            val pivotIndex = partition(nums, left, right)
            when {
                pivotIndex ==  k - 1 -> return nums[pivotIndex]
                pivotIndex < k - 1 -> left = pivotIndex + 1
                else -> right = pivotIndex - 1
            }
        }

        return -1 // If k is out of range of the array, handle accordingly
    }

    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `i` (integer), `j` (integer).
    *
    * @param nums The input array of integers.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `i` (integer), `j` (integer).
    *
    * @param nums The input array of integers.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `i` (integer), `j` (integer).
    *
    * @param nums The input array of integers.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Kth Largest Element In Array Tail Rec problem.
    * Takes `nums` (array of integers), `i` (integer), `j` (integer).
    *
    * @param nums The input array of integers.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @return Unit (no return value, modifies state in-place).
    */
    fun swap(nums: IntArray, i: Int, j: Int) {
        nums[i] = nums[j].also { nums[j] = nums[i] }
    }
}
