package linkedlist

class ReverseLinkedListIterative {

/**
 * Reverses or rearranges the input elements in place.
 *
 * @param head the root/head node of the data structure
 * @return a list/collection of result elements
 */
    fun reverseList(head: ListNode?): ListNode? {
        var prev: ListNode? = null
        var current = head
        while (current != null) {
            val nextTemp = current.next
            current.next = prev
            prev = current
            current = nextTemp
        }
        return prev
    }
}
