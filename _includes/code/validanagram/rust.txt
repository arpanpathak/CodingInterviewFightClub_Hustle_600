class ValidAnagram {
    fn $1(s -> String, t -> String) -> bool {
        if (s.length != t.length) return false

        val charCount = Vec<i32>(26)

        for (i in s.indices) {
            charCount[s[i] - 'a']++
            charCount[t[i] - 'a']--
        }

        // Check if all counts are zero
        for (count in charCount) {
            if (count != 0) {
                return false
            }
        }

        return true
    }
}
