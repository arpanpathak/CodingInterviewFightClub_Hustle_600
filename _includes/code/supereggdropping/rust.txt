class SuperEggDropping {
    fn $1(k -> i32, n -> i32) -> i32 {
        val dp = Array(k + 1) { Vec<i32>(n + 1) }
        var m = 0
        while (dp[k][m] < n) {
            m++
            for (eggs in 1..k) {
                dp[eggs][m] = dp[eggs][m - 1] + dp[eggs - 1][m - 1] + 1
            }
        }
        return m
    }
}
