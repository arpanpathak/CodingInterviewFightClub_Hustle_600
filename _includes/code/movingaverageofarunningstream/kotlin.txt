package stream


class MovingAverage(private val size: Int) {

    private val window = ArrayDeque<Int>()
    private var sum = 0

/**
 * next — executes the core logic of this algorithm on the provided input.
 *
 * @param `val` the `val` parameter — a integer value used in the computation
 * @return the computed result of type Double
 */
    fun next(`val`: Int): Double {
        window.add(`val`)
        sum += `val`
        if (window.size > size) {
            sum -= window.removeFirst()
        }

        return sum.toDouble() / window.size
    }

}
