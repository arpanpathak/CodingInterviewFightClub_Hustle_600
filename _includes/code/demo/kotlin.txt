/**
* Solves the demo problem.
* Takes `A` (array of integers).
*
* @param A The input array of integers.
* @return The computed integer result.
*/
fun solution(A: IntArray): Int {
    // Implement your solution here
    var max = A.max()

    for (i in 1..A.lastIndex)
        max = maxOf(max, A[i])

    val set = A.toSet()

    for (i in 1..max + 1)
        if (i !in set)
            return i

    return -1
}
