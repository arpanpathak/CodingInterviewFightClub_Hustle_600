package math.dp

class PascalsTriangle {

/**
 * Builds and returns a new data structure from the given input parameters.
 *
 * @param numRows the numRows parameter — a integer value used in the computation
 * @return a list/collection of result elements
 */
    fun generate(numRows: Int): List<List<Int>> {
        val result = MutableList(numRows) { MutableList(it + 1) { 1 } }

        for (i in 2 until numRows) {
            for (j in 1 until i) {
                result[i][j] = result[i - 1][j - 1] + result[i - 1][j]
            }
        }

        return result
    }
}
