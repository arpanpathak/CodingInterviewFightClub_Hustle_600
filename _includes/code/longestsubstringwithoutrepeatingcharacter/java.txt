class LongestSubstringWithoutRepeatingCharacter {
    public static void $1(s) {
        val lastIndex = IntArray(256) { -1 }
        var max = 0
        var windowStart = 0

        for (i in s.indices) {
            windowStart = maxOf(windowStart, lastIndex[s[i].code] + 1)
            max = maxOf(max, i - windowStart + 1)
            lastIndex[s[i].code] = i
        }

        return max
    }
}
