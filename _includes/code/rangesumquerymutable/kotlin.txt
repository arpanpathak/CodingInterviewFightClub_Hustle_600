package tree.fenwick

class RangeSumQueryMutable(nums: IntArray) {
    private val tree = IntArray(nums.size + 1)

    init {
        for (i in nums.indices) update(i, nums[i])
    }

    /**
    * Helper: update bit.
    *
    * @param i The integer parameter representing i.
    * @param delta The integer parameter representing delta.
    * @return Unit (no return value, modifies state in-place).
    */
    private fun updateBIT(i: Int, delta: Int) {
        var index = i + 1
        while (index <= tree.size - 1) {
            tree[index] += delta
            index += index and -index
        }
    }

    /**
    * Solves the Range Sum Query Mutable problem.
    * Takes `index` (integer), ``val`` (integer).
    *
    * @param index The integer parameter representing index.
    * @param `val` The integer parameter representing `val`.
    * @return Unit (no return value, modifies state in-place).
    */
    fun update(index: Int, `val`: Int) {
        val current = sumRange(index, index)
        updateBIT(index, `val` - current)
    }

    /**
    * Helper: sum.
    *
    * @param index The integer parameter representing index.
    * @return The computed integer result.
    */
    private fun sum(index: Int): Int {
        var i = index + 1
        var result = 0
        while (i > 0) {
            result += tree[i]
            i -= i and -i
        }
        return result
    }

    /**
    * Solves the Range Sum Query Mutable problem.
    * Takes `left` (integer), `right` (integer).
    *
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    fun sumRange(left: Int, right: Int): Int {
        return sum(right) - sum(left - 1)
    }
}
