package tree.fenwick

class RangeSumQueryMutable(nums: IntArray) {
    private val tree = IntArray(nums.size + 1)

    init {
        for (i in nums.indices) update(i, nums[i])
    }

/**
 * Updates the data structure with the provided value at the specified location.
 *
 * @param i the index position in the collection
 * @param delta the delta parameter — a integer value used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
    private fun updateBIT(i: Int, delta: Int) {
        var index = i + 1
        while (index <= tree.size - 1) {
            tree[index] += delta
            index += index and -index
        }
    }

/**
 * Updates the data structure with the provided value at the specified location.
 *
 * @param index the index position in the collection
 * @param `val` the `val` parameter — a integer value used in the computation
 * @return the computed integer result
 */
    fun update(index: Int, `val`: Int) {
        val current = sumRange(index, index)
        updateBIT(index, `val` - current)
    }

/**
 * sum — executes the core logic of this algorithm on the provided input.
 *
 * @param index the index position in the collection
 * @return the computed sum/total value
 */
    private fun sum(index: Int): Int {
        var i = index + 1
        var result = 0
        while (i > 0) {
            result += tree[i]
            i -= i and -i
        }
        return result
    }

/**
 * sum Range — executes the core logic of this algorithm on the provided input.
 *
 * @param left the left/starting boundary of the search range (inclusive)
 * @param right the right/ending boundary of the search range (inclusive)
 * @return the computed sum/total value
 */
    fun sumRange(left: Int, right: Int): Int {
        return sum(right) - sum(left - 1)
    }
}

 * Your NumArray object will be instantiated and called as such:
 * var obj = NumArray(nums)
 * obj.update(index,`val`)
 * var param_2 = obj.sumRange(left,right)
