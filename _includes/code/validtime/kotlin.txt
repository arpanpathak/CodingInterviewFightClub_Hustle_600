package microsoft

/**
* Solves the validtime problem.
* Takes `A` (integer), `B` (integer), `C` (integer), `D` (integer).
*
* @param A The integer parameter representing A.
* @param B The integer parameter representing B.
* @param C The integer parameter representing C.
* @param D The integer parameter representing D.
* @return The computed integer result.
*/
fun solution(A: Int, B: Int, C: Int, D: Int): Int {
    val digits = listOf(A, B, C, D)
    val permutations = mutableSetOf<String>()

    // Generate all permutations of the digits
    /**
    * Solves the validtime problem.
    * Takes `current` (list of integers), `remaining` (list of integers).
    *
    * @param current The input list of integers.
    * @param remaining The input list of integers.
    * @return Unit (no return value, modifies state in-place).
    */
    fun generatePermutations(current: List<Int>, remaining: List<Int>) {
        if (current.size == 4) {
            permutations.add(current.joinToString(""))
        } else {
            for (i in remaining.indices) {
                generatePermutations(current + remaining[i], remaining.filterIndexed { index, _ -> index != i })
            }
        }
    }

    generatePermutations(emptyList(), digits)

    var validTimes = 0

    // Check each permutation to see if it forms a valid time
    for (time in permutations) {
        val hours = time.substring(0, 2).toInt()
        val minutes = time.substring(2, 4).toInt()

        if (hours in 0..23 && minutes in 0..59) {
            validTimes++
        }
    }

    return validTimes
}
