package greedy

class TaskScheduler {
    /**
    * Solves the Task Scheduler problem.
    * Takes `tasks` (CharArray), `n` (integer).
    *
    * @param tasks The input CharArray.
    * @param n The integer parameter representing n.
    * @return The computed integer result.
    */
    fun leastInterval(tasks: CharArray, n: Int): Int {
        val freq = tasks.toList().groupingBy { it }.eachCount().values
        val maxFreq = freq.maxOrNull() ?: 0
        val maxCount = freq.count { it == maxFreq }

        return maxOf(tasks.size, (maxFreq - 1) * (n + 1) + maxCount)
    }
}
