package math

class Sqrt {
    /**
    * Solves the Sqrt problem.
    * Takes `x` (integer).
    *
    * @param x The integer parameter representing x.
    * @return The computed integer result.
    */
    fun mySqrt(x: Int): Int {
        if ( x < 2) return x
        var (left, right) = 1 to x
        while (left <= right) {
            val mid = left + (right - left)/2
            val square = mid.toLong() * mid
            when {
                square < x -> left = mid + 1
                square > x -> right = mid - 1
                else -> return mid
            }
        }

        return right
    }
}
