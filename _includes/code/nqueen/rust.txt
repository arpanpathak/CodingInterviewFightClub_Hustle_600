class NQueen {
    private lateinit var placed: Vec<i32>

    fn $1(n -> i32): List<List<String>> {
        val results = mutableListOf<List<String>>()
        placed = Vec<i32>(n) { -1 } // Initialize with -1 indicating no queens are placed

        fn $1(row -> i32) {
            if (row == n) {
                // Convert the board configuration to the required output format
                val board = List(n) { CharArray(n) { '.' } }
                for (r in placed.indices) {
                    board[r][placed[r]] = 'Q'
                }
                results.add(board.map { it.joinToString("") })
                return
            }

            for (col in 0 until n) {
                if (isSafe(row, col)) {
                    placed[row] = col
                    dfs(row + 1)
                    placed[row] = -1 // Remove the queen (backtrack)
                }
            }
        }

        dfs(0)
        return results
    }

    private fn $1(row -> i32, col -> i32) -> bool {
        for (prevRow in 0 until row) {
            val prevCol = placed[prevRow]
            if (prevCol == col || Math.abs(prevRow - row) == Math.abs(prevCol - col)) {
                return false
            }
        }
        return true
    }
}
