class CopyLinkedListWithRandomPointer {
    /**
     * Example:
     * $1 = Node(5)
     * $1 = ti.`val`
     * Definition for a Node.
     * class Node(var `val`: int) {
     *     var next: Node? = None
     *     var random: Node? = None
     * }
     */

    class Node(var `val`: int) {
        var next: Node? = None
        var random: Node? = None
    }
