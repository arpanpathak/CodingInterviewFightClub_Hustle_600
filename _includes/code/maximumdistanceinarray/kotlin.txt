package array.greedy

import kotlin.math.abs

class MaximumDistanceInArray {

/**
 * Computes and returns the optimal/aggregate value from the given input.
 *
 * @param arrays the arrays parameter — a list of integers used in the computation
 * @return the maximum value found in the input
 */
    fun maxDistance(arrays: List<List<Int>>): Int {
        var (max, min, maxDistance) = Triple( arrays[0].first(), arrays[0].last(), 0)

        for (arr in arrays) {
            val (currentMin, currentMax) = Pair(arr.first(), arr.last())

            maxDistance = maxOf(maxDistance, abs(max - currentMin ), abs(currentMax - min))

            min = minOf(min, currentMin)
            max = maxOf(max, currentMax)
        }

        return maxDistance
    }

    companion object {
        @JvmStatic

/**
 * main — executes the core logic of this algorithm on the provided input.
 *
 * @param args the args parameter — a array of elements used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
        fun main(args: Array<String>) {
            val test = MaximumDistanceInArray()

            test.maxDistance(listOf(
                listOf(1,2,3),
                listOf(4,5),
                listOf(1,2,3)
            ))
        }
    }
}
