package array.greedy

class MinimumNumberofSwapstoMaketheStringBalanced {

/**
 * Computes and returns the optimal/aggregate value from the given input.
 *
 * @param s the input string to process
 * @return the minimum value found in the input
 */
    fun minSwaps(s: String): Int {
        var imbalance = 0
        var maxImbalance = 0

        for (char in s) {
            if (char == '[') {
                imbalance--
            } else { // char == ']'
                imbalance++
            }
            maxImbalance = maxOf(maxImbalance, imbalance)
        }

        return (maxImbalance + 1) / 2
    }
}
