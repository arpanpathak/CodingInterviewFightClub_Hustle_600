class MaximumSumSubArray {
    // Kaden's Algorithm Dynamic Programming
    $1(nums: vector<int>) {
        if (nums.isEmpty())
            return 0
        var (maxSum, currentSum) = listOf( nums[0], nums[0] )


        for (i in 1 until nums.size) {
            currentSum = max(nums[i], currentSum + nums[i])
            maxSum = max(maxSum, currentSum)
        }

        return maxSum
    }
}
