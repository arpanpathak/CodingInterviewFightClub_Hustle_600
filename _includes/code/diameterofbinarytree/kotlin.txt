package graph

import javax.swing.Spring.height

class DiameterOfBinaryTree {
    /**
    * Solves the Diameter Of Binary Tree problem.
    * Takes `root` (binary tree node reference).
    *
    * @param root The binary tree node reference (nullable).
    * @return The computed integer result.
    */
    fun diameterOfBinaryTree(root: TreeNode?): Int {
        var max = 0
        /**
        * Solves the Diameter Of Binary Tree problem.
        * Takes `root` (binary tree node reference).
        *
        * @param root The binary tree node reference (nullable).
        * @return The computed integer result.
        */
        fun height(root: TreeNode?): Int {
            if (null == root) return -1
            val left = height(root.left)
            val right = height(root.right)
            max = Math.max(max, 2 + left + right)
            return 1 + Math.max(left, right)
        }

        height(root)
        return max
    }
}
