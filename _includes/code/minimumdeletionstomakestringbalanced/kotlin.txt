package stack

class MinimumDeletionsToMakeStringBalanced {

/**
 * Computes and returns the optimal/aggregate value from the given input.
 *
 * @param s the input string to process
 * @return the minimum value found in the input
 */
    fun minimumDeletions(s: String): Int {
        val stack = ArrayDeque<Char>()
        var deletions = 0

        for (ch in s) {
            if (ch == 'b') {
                stack.addLast('b')
            } else { // ch == 'a'
                if (stack.isNotEmpty() && stack.last() == 'b') {
                    stack.removeLast()      // delete 'b'
                    deletions++      // count this deletion
                }
            }
        }

        return deletions
    }
}
