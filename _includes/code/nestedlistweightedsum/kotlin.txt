class NestedListWeightedSum {
    /**
    * Solves the Nested Integer problem.
    * Takes `nestedList` (List<NestedInteger>).
    *
    * @param nestedList The input List<NestedInteger>.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `nestedList` (List<NestedInteger>).
    *
    * @param nestedList The input List<NestedInteger>.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `nestedList` (List<NestedInteger>).
    *
    * @param nestedList The input List<NestedInteger>.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `nestedList` (List<NestedInteger>).
    *
    * @param nestedList The input List<NestedInteger>.
    * @return The computed integer result.
    */
    fun depthSum(nestedList: List<NestedInteger>): Int {
        return dfs(nestedList, 1)
    }

    /**
    * Solves the Nested Integer problem.
    * Takes `list` (List<NestedInteger>), `depth` (integer).
    *
    * @param list The input List<NestedInteger>.
    * @param depth The integer parameter representing depth.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `list` (List<NestedInteger>), `depth` (integer).
    *
    * @param list The input List<NestedInteger>.
    * @param depth The integer parameter representing depth.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `list` (List<NestedInteger>), `depth` (integer).
    *
    * @param list The input List<NestedInteger>.
    * @param depth The integer parameter representing depth.
    * @return The computed integer result.
    */
    /**
    * Solves the Nested Integer problem.
    * Takes `list` (List<NestedInteger>), `depth` (integer).
    *
    * @param list The input List<NestedInteger>.
    * @param depth The integer parameter representing depth.
    * @return The computed integer result.
    */
    fun dfs(list: List<NestedInteger>, depth: Int): Int {
        var sum = 0
        for (nested in list) {
            if (nested.isInteger()) {
                sum += (depth * nested.getInteger() !!)
            } else {
                sum += dfs(nested.getList()!!, depth + 1)
            }
        }

        return sum
    }
}
