package array.prefixsum

class SubArrayProductLessThanK {

/**
 * num Subarray Product Less Than K — executes the core logic of this algorithm on the provided input.
 *
 * @param nums the input array of numbers to process
 * @param k the number of elements/operations to consider (k parameter)
 * @return the total count/number of matching elements
 */
    fun numSubarrayProductLessThanK(nums: IntArray, k: Int): Int {
        if ( k <=1) return 0

        var (count, product, left) = listOf(0, 1, 0)

        for (right in nums.indices) {
            product *= nums[right]

            while (product >=k ) {
                product /= nums[left++]
            }

            count += (right - left) + 1
        }

        return count
    }
}
