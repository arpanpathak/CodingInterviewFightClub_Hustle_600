    private val dp = mutableMapOf<Triple<Int, Int, Int>, Int>()
    private lateinit var grid: Array<IntArray>
    private var n: Int = 0

    /**
    * Solves the Cherry Pickup problem.

    /**

    /**
    * Solves the Cherry Pickup problem.
    * Takes `grid` (2D matrix of integers).
    *
    * @param grid The input 2D matrix of integers.
    * @return The computed integer result.
    */
    fun cherryPickup(grid: Array<IntArray>): Int {
        this.grid = grid
        n = grid.size
        dp.clear()
        return maxOf(0, dfs(0, 0, 0))
    }

    /**
    * Helper: dfs.
    *
    * @param row The integer parameter representing row.
    * @param col1 The integer parameter representing col1.

    */
    /**
    * Helper: dfs.

    * @param col2 The integer parameter representing col2.
    * @return The computed integer result.
    */
    /**
    * Helper: dfs.
    *
    * @param row The integer parameter representing row.
    * @param col1 The integer parameter representing col1.
    * @param col2 The integer parameter representing col2.
    * @return The computed integer result.
    */
    /**
    * Helper: dfs.
    *
    * @param row The integer parameter representing row.
    * @param col1 The integer parameter representing col1.
    * @param col2 The integer parameter representing col2.
    * @return The computed integer result.
    */
    private fun dfs(row: Int, col1: Int, col2: Int): Int {
        val row2 = row + col1 - col2
        val state = Triple(row, col1, col2)

        return when {
            dp.contains(state) -> dp[state] !!
            isInvalidPosition(row, row2, col1, col2) -> Int.MIN_VALUE
            isFinalPosition(row, col1) -> grid[row][col1]
            else -> {
                val cherryCount = when {
                    (col1 != col2) -> grid[row][col1] + grid[row2][col2]
                    else ->  grid[row][col1]
                }
                val maxCherries = maxOf(
                    dfs(row, col1 + 1, col2 + 1),
                    dfs(row, col1 + 1, col2),
                    dfs(row + 1, col1, col2),
                    dfs(row + 1, col1, col2 + 1)
                )
                (cherryCount + maxCherries)
            }
        }.also { dp[state] = it }
    }
