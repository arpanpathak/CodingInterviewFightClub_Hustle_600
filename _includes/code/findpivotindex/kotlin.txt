package array.prefixsum

class FindPivotIndex {

/**
 * pivot Index — executes the core logic of this algorithm on the provided input.
 *
 * @param nums the input array of numbers to process
 * @return the computed index/position in the array
 */
    fun pivotIndex(nums: IntArray): Int {
        val totalSum = nums.sum()
        var prefixSum = 0

        for (i in nums.indices ) {
            if (prefixSum == totalSum - prefixSum - nums[i])
                return i

            prefixSum += nums[i]
        }

        return -1
    }
}
