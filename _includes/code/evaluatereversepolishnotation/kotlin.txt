package stack

class EvaluateReversePolishNotation {
    private val SYMBOLS = setOf("+", "-", "*", "/")

    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `tokens` (Array<String>).
    *
    * @param tokens The input Array<String>.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `tokens` (Array<String>).
    *
    * @param tokens The input Array<String>.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `tokens` (Array<String>).
    *
    * @param tokens The input Array<String>.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `tokens` (Array<String>).
    *
    * @param tokens The input Array<String>.
    * @return The computed integer result.
    */
    fun evalRPN(tokens: Array<String>): Int {
        val stack = mutableListOf<Int>()
        for (token in tokens) {
            if (token in SYMBOLS) {
                val op2 = stack.removeLast()
                val op1 = stack.removeLast()
                stack.add(operate(token[0], op1, op2))
            } else
                stack.add(token.toInt())
        }
        return stack.removeLast()
    }

    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `symbol` (character), `a` (integer), `b` (integer).
    *
    * @param symbol The character.
    * @param a The integer parameter representing a.
    * @param b The integer parameter representing b.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `symbol` (character), `a` (integer), `b` (integer).
    *
    * @param symbol The character.
    * @param a The integer parameter representing a.
    * @param b The integer parameter representing b.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `symbol` (character), `a` (integer), `b` (integer).
    *
    * @param symbol The character.
    * @param a The integer parameter representing a.
    * @param b The integer parameter representing b.
    * @return The computed integer result.
    */
    /**
    * Solves the Evaluate Reverse Polish Notation problem.
    * Takes `symbol` (character), `a` (integer), `b` (integer).
    *
    * @param symbol The character.
    * @param a The integer parameter representing a.
    * @param b The integer parameter representing b.
    * @return The computed integer result.
    */
    fun operate(symbol: Char, a: Int, b: Int): Int {
        return when(symbol) {
            '+' -> a + b
            '-' -> a - b
            '*' -> a * b
            '/' -> a / b
            else -> 0
        }
    }
}
