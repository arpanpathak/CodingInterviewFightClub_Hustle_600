package array.dp

class LongestCommonSubarray {
    val memo = mutableMapOf<String, Int>()

    /**
    * Solves the Longest Common Subarray problem.
    * Takes `nums1` (array of integers), `nums2` (array of integers).
    *
    * @param nums1 The input array of integers.
    * @param nums2 The input array of integers.
    * @return The computed integer result.
    */
    fun findLength(nums1: IntArray, nums2: IntArray): Int {
        return dp(nums1, nums2, nums1.lastIndex, nums2.lastIndex, Int.MIN_VALUE)
    }

    /**
    * Solves the Longest Common Subarray problem.
    * Takes `nums1` (array of integers), `nums2` (array of integers), `i` (integer), `j` (integer), `max` (integer).
    *
    * @param nums1 The input array of integers.
    * @param nums2 The input array of integers.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @param max The integer parameter representing max.
    * @return The computed integer result.
    */
    fun dp(nums1: IntArray, nums2: IntArray, i: Int, j: Int, max: Int): Int {
        if (i == 0 || j == 0)
            return max
        val key ="$i.$j"
        if (memo.containsKey(key))
            return memo[key]!!

        var maxSoFar = max

        if (nums1[i] == nums1[j])
            maxSoFar = dp(nums1, nums2, i + 1, j + 1, max + 1)

        return maxSoFar
    }

    /**
    * Solves the Longest Common Subarray problem.
    * Takes `nums1` (array of integers), `nums2` (array of integers).
    *
    * @param nums1 The input array of integers.
    * @param nums2 The input array of integers.
    * @return The computed integer result.
    */
    fun findLength_tabular(nums1: IntArray, nums2: IntArray): Int {
        val m = nums1.size
        val n = nums2.size
        val dp = Array(m + 1) { IntArray(n + 1) }
        var maxLength = 0

        // Fill the dp array
        for (i in 1..m) {
            for (j in 1..n) {
                if (nums1[i - 1] == nums2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1
                    maxLength = maxOf(maxLength, dp[i][j])
                } else {
                    dp[i][j] = 0
                }
            }
        }

        return maxLength
    }
}
