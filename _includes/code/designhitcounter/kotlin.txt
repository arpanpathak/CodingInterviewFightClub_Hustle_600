package queueu.dequeue

import java.util.*

class HitCounter() {
    private val hits: Deque<Int> = LinkedList()

/**
 * hit — executes the core logic of this algorithm on the provided input.
 *
 * @param timestamp the timestamp parameter — a integer value used in the computation
 * @return the computed integer result
 */
    fun hit(timestamp: Int) {
        hits.offer(timestamp)
    }

/**
 * Retrieves and returns the requested element or value from the data structure.
 *
 * @param timestamp the timestamp parameter — a integer value used in the computation
 * @return the computed integer result
 */
    fun getHits(timestamp: Int): Int {
        while (hits.isNotEmpty() && hits.peekFirst() <= timestamp - 300) {
            hits.pollFirst()
        }
        return hits.size
    }
}
