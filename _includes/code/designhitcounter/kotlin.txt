package queueu.dequeue

import java.util.*

class HitCounter() {
    private val hits: Deque<Int> = LinkedList()

    /**
    * Solves the Hit Counter problem.
    * Takes `timestamp` (integer).
    *
    * @param timestamp The integer parameter representing timestamp.
    * @return Unit (no return value, modifies state in-place).
    */
    fun hit(timestamp: Int) {
        hits.offer(timestamp)
    }

    /**
    * Solves the Hit Counter problem.
    * Takes `timestamp` (integer).
    *
    * @param timestamp The integer parameter representing timestamp.
    * @return The computed integer result.
    */
    fun getHits(timestamp: Int): Int {
        while (hits.isNotEmpty() && hits.peekFirst() <= timestamp - 300) {
            hits.pollFirst()
        }
        return hits.size
    }
}
