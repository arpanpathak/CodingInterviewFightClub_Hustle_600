package sorting

import oracle.net.aso.j

class MergeSort {

/**
 * Sorts or reorders the input elements according to the specified criteria.
 *
 * @param A the first input array for comparison/merging
 * @param L the L parameter — a array of integers used in the computation
 * @param leftCount the leftCount parameter — a integer value used in the computation
 * @param R the R parameter — a array of integers used in the computation
 * @param rightCount the rightCount parameter — a integer value used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
    fun merge(A: IntArray, L: IntArray, leftCount: Int, R: IntArray, rightCount: Int) {
        var (i, j, k) = listOf(0, 0, 0)

        while (i < leftCount && j < rightCount) {
            if (L[i] < R[j]) A[k++] = L[i++]
            else A[k++] = R[j++]
        }
        while (i < leftCount) A[k++] = L[i++]
        while (j < rightCount) A[k++] = R[j++]
    }

/**
 * Sorts or reorders the input elements according to the specified criteria.
 *
 * @param arr the input array of numbers to process
 * @param n the size/dimension parameter for the algorithm
 * @return Unit (nothing) — this function operates via side effects
 */
    fun merge_sort(arr: IntArray, n: Int) {
        if (n < 2) return
        val mid = n / 2
        val L = IntArray(mid) { arr[it] }
        val R = IntArray(n - mid) { arr[it + mid] }
        merge_sort(L, mid)
        merge_sort(R, n - mid)
        merge(arr, L, mid, R, n - mid)
    }
}
