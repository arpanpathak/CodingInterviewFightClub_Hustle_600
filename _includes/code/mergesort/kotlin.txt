package sorting

import oracle.net.aso.j

class MergeSort {
    /**
    * Solves the Merge Sort problem.
    * Takes `A` (array of integers), `L` (array of integers), `leftCount` (integer), `R` (array of integers), `rightCount` (integer).
    *
    * @param A The input array of integers.
    * @param L The input array of integers.
    * @param leftCount The integer parameter representing leftCount.
    * @param R The input array of integers.
    * @param rightCount The integer parameter representing rightCount.
    * @return Unit (no return value, modifies state in-place).
    */
    fun merge(A: IntArray, L: IntArray, leftCount: Int, R: IntArray, rightCount: Int) {
        var (i, j, k) = listOf(0, 0, 0)

        while (i < leftCount && j < rightCount) {
            if (L[i] < R[j]) A[k++] = L[i++]
            else A[k++] = R[j++]
        }
        while (i < leftCount) A[k++] = L[i++]
        while (j < rightCount) A[k++] = R[j++]
    }

    /**
    * Solves the Merge Sort problem.
    * Takes `arr` (array of integers), `n` (integer).
    *
    * @param arr The input array of integers.
    * @param n The integer parameter representing n.
    * @return Unit (no return value, modifies state in-place).
    */
    fun merge_sort(arr: IntArray, n: Int) {
        if (n < 2) return
        val mid = n / 2
        val L = IntArray(mid) { arr[it] }
        val R = IntArray(n - mid) { arr[it + mid] }
        merge_sort(L, mid)
        merge_sort(R, n - mid)
        merge(arr, L, mid, R, n - mid)
    }
}
