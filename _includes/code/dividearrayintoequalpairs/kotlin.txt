package array.hashtable

class DivideArrayIntoEqualPairs {

/**
 * divide Array — executes the core logic of this algorithm on the provided input.
 *
 * @param nums the input array of numbers to process
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
    fun divideArray(nums: IntArray): Boolean {
        val freqMap = mutableMapOf<Int, Int>()

        // Count the frequency of each number
        for (num in nums) {
            freqMap[num] = freqMap.getOrDefault(num, 0) + 1
        }

        // Check if every number has an even frequency
        for (count in freqMap.values) {
            if (count % 2 != 0) {
                return false
            }
        }

        return true
    }
}
