class TwoSum_II {
    public static void $1(numbers: IntArray, target): IntArray {
        var start = 0
        var end = numbers.lastIndex

        while ( start < end ) {
            val sum = numbers[start] + numbers[end]

            when {
                sum == target -> {
                    return intArrayOf(start + 1, end + 1)
                }
                sum < target -> start++
                else -> end--
            }
        }

        return intArrayOf()
    }
}
