package queueu.dequeue

class DesignACircularQueue {
    class MyCircularQueue(k: Int) {
        private val queue = IntArray(k)
        private var front = 0
        private var rear = 0
        private var size = 0
        private val capacity = k

        /**
        * Solves the Design ACircular Queue problem.
        * Takes `value` (integer).
        *
        * @param value The integer parameter representing value.
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        * Takes `value` (integer).
        *
        * @param value The integer parameter representing value.
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        * Takes `value` (integer).
        *
        * @param value The integer parameter representing value.
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        * Takes `value` (integer).
        *
        * @param value The integer parameter representing value.
        * @return `true` if the condition is met, `false` otherwise.
        */
        fun enQueue(value: Int): Boolean {
            if (isFull()) {
                return false
            }
            queue[rear] = value
            rear = (rear + 1) % capacity
            size++
            return true
        }

        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        fun deQueue(): Boolean {
            if (isEmpty()) {
                return false
            }
            front = (front + 1) % capacity
            size--
            return true
        }

        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        fun Front(): Int {
            if (isEmpty()) {
                return -1
            }
            return queue[front]
        }

        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return The computed integer result.
        */
        fun Rear(): Int {
            if (isEmpty()) {
                return -1
            }
            return queue[(rear - 1 + capacity) % capacity]
        }

        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        fun isEmpty(): Boolean {
            return size == 0
        }

        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        /**
        * Solves the Design ACircular Queue problem.
        *
        * @return `true` if the condition is met, `false` otherwise.
        */
        fun isFull(): Boolean {
            return size == capacity
        }
    }
}
