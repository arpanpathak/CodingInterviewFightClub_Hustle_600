package queueu.dequeue

class DesignACircularQueue {
    class MyCircularQueue(k: Int) {
        private val queue = IntArray(k)
        private var front = 0
        private var rear = 0
        private var size = 0
        private val capacity = k

/**
 * en Queue — executes the core logic of this algorithm on the provided input.
 *
 * @param value the target value to search for or match against
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
        fun enQueue(value: Int): Boolean {
            if (isFull()) {
                return false
            }
            queue[rear] = value
            rear = (rear + 1) % capacity
            size++
            return true
        }

/**
 * de Queue — executes the core logic of this algorithm on the provided input.
 *
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
        fun deQueue(): Boolean {
            if (isEmpty()) {
                return false
            }
            front = (front + 1) % capacity
            size--
            return true
        }

/**
 * Front — executes the core logic of this algorithm on the provided input.
 *
 * @return the computed integer result
 */
        fun Front(): Int {
            if (isEmpty()) {
                return -1
            }
            return queue[front]
        }

/**
 * Rear — executes the core logic of this algorithm on the provided input.
 *
 * @return the computed integer result
 */
        fun Rear(): Int {
            if (isEmpty()) {
                return -1
            }
            return queue[(rear - 1 + capacity) % capacity]
        }

/**
 * is Empty — executes the core logic of this algorithm on the provided input.
 *
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
        fun isEmpty(): Boolean {
            return size == 0
        }

/**
 * is Full — executes the core logic of this algorithm on the provided input.
 *
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
        fun isFull(): Boolean {
            return size == capacity
        }
    }
}
