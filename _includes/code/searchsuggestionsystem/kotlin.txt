package trie

import java.lang.StringBuilder
import java.util.*


class Trie {
    data class TrieNode(
        var ch: Char='*',
        var isWord: Boolean = false,
        var children: TreeMap<Char, TrieNode> = TreeMap()
    )

    val root = TrieNode()

    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return Unit (no return value, modifies state in-place).
    */
    fun insert(word: String) {
        var currentNode = root

        word.forEach { ch->  currentNode = currentNode.children.getOrPut(ch){ TrieNode(ch) } }

        currentNode.isWord = true
    }

    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return The resulting collection (list of strings).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return The resulting collection (list of strings).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return The resulting collection (list of strings).
    */
    /**
    * Solves the Trie problem.
    * Takes `word` (string).
    *
    * @param word The input string.
    * @return The resulting collection (list of strings).
    */
    fun search(word: String): List<String> {
        val results = mutableListOf<String>()
        var currentNode = root

        word.forEach { ch -> currentNode = currentNode.children[ch] ?: return results }
        // If we reach here, currentNode represents the node ending at 'word'
        // Collect all words from this node downwards
        collectWords(currentNode, word, results)

        return results
    }

    /**
    * Solves the Trie problem.
    * Takes `node` (TrieNode?), `prefix` (string), `results` (MutableList<String>).
    *
    * @param node The TrieNode? (nullable).
    * @param prefix The input string.
    * @param results The input MutableList<String>.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `node` (TrieNode?), `prefix` (string), `results` (MutableList<String>).
    *
    * @param node The TrieNode? (nullable).
    * @param prefix The input string.
    * @param results The input MutableList<String>.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `node` (TrieNode?), `prefix` (string), `results` (MutableList<String>).
    *
    * @param node The TrieNode? (nullable).
    * @param prefix The input string.
    * @param results The input MutableList<String>.
    * @return Unit (no return value, modifies state in-place).
    */
    /**
    * Solves the Trie problem.
    * Takes `node` (TrieNode?), `prefix` (string), `results` (MutableList<String>).
    *
    * @param node The TrieNode? (nullable).
    * @param prefix The input string.
    * @param results The input MutableList<String>.
    * @return Unit (no return value, modifies state in-place).
    */
    fun collectWords(node: TrieNode?, prefix: String, results: MutableList<String>) {
        if (node == null)
            return
        if (node.isWord)
            results.add(prefix)

        for((ch, childNode) in node.children) {
            collectWords(childNode, prefix + ch, results)
        }
    }
}

class SearchSuggestionSystem {
    /**
    * Solves the Trie problem.
    * Takes `products` (Array<String>), `searchWord` (string).
    *
    * @param products The input Array<String>.
    * @param searchWord The input string.
    * @return The resulting collection (List<List<String>).
    */
    /**
    * Solves the Trie problem.
    * Takes `products` (Array<String>), `searchWord` (string).
    *
    * @param products The input Array<String>.
    * @param searchWord The input string.
    * @return The resulting collection (List<List<String>).
    */
    /**
    * Solves the Trie problem.
    * Takes `products` (Array<String>), `searchWord` (string).
    *
    * @param products The input Array<String>.
    * @param searchWord The input string.
    * @return The resulting collection (List<List<String>).
    */
    /**
    * Solves the Trie problem.
    * Takes `products` (Array<String>), `searchWord` (string).
    *
    * @param products The input Array<String>.
    * @param searchWord The input string.
    * @return The resulting collection (List<List<String>).
    */
    fun suggestedProducts(products: Array<String>, searchWord: String): List<List<String>> {
        val trie = Trie()
        val result = mutableListOf<List<String>>()
        val prefixSearchString = StringBuilder()

        products.forEach { trie.insert(it) }

        searchWord.forEach { ch ->
            prefixSearchString.append(ch)
            result.add(trie.search(prefixSearchString.toString()).take(3))
        }

        return  result
    }
}
