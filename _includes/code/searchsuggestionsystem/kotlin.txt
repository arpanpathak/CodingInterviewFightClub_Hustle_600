package trie

import java.lang.StringBuilder
import java.util.*


class Trie {
    data class TrieNode(
        var ch: Char='*',
        var isWord: Boolean = false,
        var children: TreeMap<Char, TrieNode> = TreeMap()
    )

    val root = TrieNode()

/**
 * Inserts the specified element into the data structure.
 *
 * @param word the word parameter — a string value used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
    fun insert(word: String) {
        var currentNode = root

        word.forEach { ch->  currentNode = currentNode.children.getOrPut(ch){ TrieNode(ch) } }

        currentNode.isWord = true
    }

/**
 * Searches for and returns the target element/position using an efficient algorithm.
 *
 * @param word the word parameter — a string value used in the computation
 * @return a list of matching/found elements
 */
    fun search(word: String): List<String> {
        val results = mutableListOf<String>()
        var currentNode = root

        word.forEach { ch -> currentNode = currentNode.children[ch] ?: return results }
        // If we reach here, currentNode represents the node ending at 'word'
        // Collect all words from this node downwards
        collectWords(currentNode, word, results)

        return results
    }

/**
 * collect Words — executes the core logic of this algorithm on the provided input.
 *
 * @param node the root/head node of the data structure
 * @param prefix the prefix parameter — a string value used in the computation
 * @param results the results parameter — a list of strings used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
    fun collectWords(node: TrieNode?, prefix: String, results: MutableList<String>) {
        if (node == null)
            return
        if (node.isWord)
            results.add(prefix)

        for((ch, childNode) in node.children) {
            collectWords(childNode, prefix + ch, results)
        }
    }
}

class SearchSuggestionSystem {

/**
 * suggested Products — executes the core logic of this algorithm on the provided input.
 *
 * @param products the products parameter — a array of elements used in the computation
 * @param searchWord the searchWord parameter — a string value used in the computation
 * @return a list/collection of result elements
 */
    fun suggestedProducts(products: Array<String>, searchWord: String): List<List<String>> {
        val trie = Trie()
        val result = mutableListOf<List<String>>()
        val prefixSearchString = StringBuilder()

        products.forEach { trie.insert(it) }

        searchWord.forEach { ch ->
            prefixSearchString.append(ch)
            result.add(trie.search(prefixSearchString.toString()).take(3))
        }

        return  result
    }
}
