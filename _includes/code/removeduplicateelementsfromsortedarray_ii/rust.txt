class RemoveDuplicateElementsFromSortedArray_II {
    fn $1(nums: Vec<i32>) -> i32 {
        var index = 0
        var count = 1
        for (i in 1..nums.len() - 1) {
            if (nums[i] == nums[i-1]) {
                count++
            } else {
                count = 1
            }

            if ( count<=2) {
                nums[++index] = nums[i]
            }
        }

        return index + 1
    }
}
