package stack

class MinStack() {
    private val stack = ArrayDeque<Int>()
    private val minStack = ArrayDeque<Int>()

    /**
    * Solves the Min Stack problem.
    * Takes ``val`` (integer).
    *
    * @param `val` The integer parameter representing `val`.
    * @return Unit (no return value, modifies state in-place).
    */
    fun push(`val`: Int) {
        stack.add(`val`)
        if (minStack.isEmpty() || `val` <= minStack.last()) {
            minStack.add(`val`)
        }
    }

    /**
    * Solves the Min Stack problem.
    *
    * @return Unit (no return value, modifies state in-place).
    */
    fun pop() {
        if (stack.isNotEmpty()) {
            val poppedValue = stack.removeLast()
            if (poppedValue == minStack.last()) {
                minStack.removeLast()
            }
        }
    }

    /**
    * Solves the Min Stack problem.
    *
    * @return The computed integer result.
    */
    fun top(): Int {
        return stack.last()
    }

    /**
    * Solves the Min Stack problem.
    *
    * @return The computed integer result.
    */
    fun getMin(): Int {
        return minStack.last()
    }
}
