package stock_market.dp

class BestTimeToBuyAndSellStockWithTransactionFee {

    /**
    * Solves the Best Time To Buy And Sell Stock With Transaction Fee problem.
    * Takes `prices` (array of integers), `fee` (integer).
    *
    * @param prices The input array of integers.
    * @param fee The integer parameter representing fee.
    * @return The computed integer result.
    */
    /**
    * Solves the Best Time To Buy And Sell Stock With Transaction Fee problem.
    * Takes `prices` (array of integers), `fee` (integer).
    *
    * @param prices The input array of integers.
    * @param fee The integer parameter representing fee.
    * @return The computed integer result.
    */
    /**
    * Solves the Best Time To Buy And Sell Stock With Transaction Fee problem.
    * Takes `prices` (array of integers), `fee` (integer).
    *
    * @param prices The input array of integers.
    * @param fee The integer parameter representing fee.
    * @return The computed integer result.
    */
    /**
    * Solves the Best Time To Buy And Sell Stock With Transaction Fee problem.
    * Takes `prices` (array of integers), `fee` (integer).
    *
    * @param prices The input array of integers.
    * @param fee The integer parameter representing fee.
    * @return The computed integer result.
    */
    fun maxProfit(prices: IntArray, fee: Int): Int {
        if (prices.isEmpty()) return 0

        var hold = -prices[0]
        var cash = 0

        for (i in 1..prices.lastIndex) {
            cash = maxOf( cash, hold + prices[i] - fee) // Sell the stock
            hold = maxOf(hold, cash - prices[i]) // buy the stock
        }

        return cash
    }
}
