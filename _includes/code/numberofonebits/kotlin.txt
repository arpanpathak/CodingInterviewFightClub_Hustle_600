package bitset

class NumberOfOneBits {
    /**
    * Solves the Number Of One Bits problem.
    * Takes `n` (integer).
    *
    * @param n The integer parameter representing n.
    * @return The computed integer result.
    */
    fun hammingWeight(n: Int): Int {
        var number = n
        var count = 0

        while (number > 0) {
            number = number and (number - 1)
            count++
        }

        return count
    }
}
