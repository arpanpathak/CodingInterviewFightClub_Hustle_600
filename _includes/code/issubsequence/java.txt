class IsSubsequence {
    public static void $1(s, t) {
        var i = 0
        var j = 0

        while (i < s.length && j < t.length) {
            if (s[i] == t[j]) {
                i++
                j++
            } else {
                j++
            }
        }

        // Match found
        return i == s.length
    }
}
