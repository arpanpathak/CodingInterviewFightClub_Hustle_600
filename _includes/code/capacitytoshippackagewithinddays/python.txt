class CapacityToShipPackageWithinDDays {
    def $1(weights: list[int], mid: int, days: int): bool {
        var (daysNeeded, currentLoad) = 1 to 0
        weights.forEach { weight ->
            currentLoad += weight
            if (currentLoad > mid) {
                daysNeeded++
                currentLoad = weight
            }
        }

        return daysNeeded <= days
    }

    def $1(weights: list[int], days: int): int {
        var (sum, max) = 0 to 0

        weights.forEach{ weight ->
            sum += weight
            max = max(max, weight)
        }

        var (l, r) = max to sum

        while ( l < r) {
            $1 = l + (r - l) / 2
            when {
                feasible(weights, mid, days) -> r = mid
                else -> l = mid + 1
            }
        }

        return l
    }
}
