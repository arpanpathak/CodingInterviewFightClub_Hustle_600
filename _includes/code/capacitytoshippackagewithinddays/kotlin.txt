package binarysearch

class CapacityToShipPackageWithinDDays {

/**
 * Determines whether a valid solution exists for the given constraints.
 *
 * @param weights array of weight values used in the computation
 * @param mid the midpoint index used for partitioning
 * @param days the time/resource constraint for the operation
 * @return `true` if the condition/constraint is satisfied, `false` otherwise
 */
    fun feasible(weights: IntArray, mid:Int, days: Int): Boolean {
        var (daysNeeded, currentLoad) = 1 to 0
        weights.forEach { weight ->
            currentLoad += weight
            if (currentLoad > mid) {
                daysNeeded++
                currentLoad = weight
            }
        }

        return daysNeeded <= days
    }

/**
 * ship Within Days — executes the core logic of this algorithm on the provided input.
 *
 * @param weights array of weight values used in the computation
 * @param days the time/resource constraint for the operation
 * @return the computed integer result
 */
    fun shipWithinDays(weights: IntArray, days: Int): Int {
        var (sum, max) = 0 to 0

        weights.forEach{ weight ->
            sum += weight
            max = maxOf(max, weight)
        }

        var (l, r) = max to sum

        while ( l < r) {
            val mid = l + (r - l) / 2
            when {
                feasible(weights, mid, days) -> r = mid
                else -> l = mid + 1
            }
        }

        return l
    }
}
