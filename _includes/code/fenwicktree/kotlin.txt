package tree.fenwick

class FenwickTree(size: Int) {
    private val tree = IntArray(size + 1)

    /**
    * Solves the Fenwick Tree problem.
    * Takes `index` (integer), `value` (integer).
    *
    * @param index The integer parameter representing index.
    * @param value The integer parameter representing value.
    * @return Unit (no return value, modifies state in-place).
    */
    fun update(index: Int, value: Int) {
        var i = index + 1
        while (i < tree.size) {
            tree[i] += value
            i += i and -i
        }
    }

    /**
    * Solves the Fenwick Tree problem.
    * Takes `index` (integer).
    *
    * @param index The integer parameter representing index.
    * @return The computed integer result.
    */
    fun prefixSum(index: Int): Int {
        var i = index + 1
        var sum = 0
        while (i > 0) {
            sum += tree[i]
            i -= i and -i
        }
        return sum
    }

    /**
    * Solves the Fenwick Tree problem.
    * Takes `left` (integer), `right` (integer).
    *
    * @param left The integer parameter representing left.
    * @param right The integer parameter representing right.
    * @return The computed integer result.
    */
    fun rangeSum(left: Int, right: Int): Int = prefixSum(right) - prefixSum(left - 1)
}
