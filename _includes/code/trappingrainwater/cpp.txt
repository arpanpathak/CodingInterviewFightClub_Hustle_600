    $1(height: vector<int>) {
        var (left, right, leftMax, rightMax, waterTrapped) = listOf(0, height.size() - 1, 0, 0, 0)

        while (left <= right) {
            if (height[left] <= height[right]) {
                waterTrapped += (leftMax - height[left]).coerceAtLeast(0)
                leftMax = max(leftMax, height[left])

            }
        }
