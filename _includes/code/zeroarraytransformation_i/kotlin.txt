    /**
    * Solves the Zero Array Transformation_I problem.
    * Takes `nums` (array of integers), `queries` (2D matrix of integers).
    *
    * @param nums The input array of integers.
    * @param queries The input 2D matrix of integers.
    * @return `true` if the condition is met, `false` otherwise.
    */
    /**
    * Solves the Zero Array Transformation_I problem.
    * Takes `nums` (array of integers), `queries` (2D matrix of integers).
    *
    * @param nums The input array of integers.
    * @param queries The input 2D matrix of integers.
    * @return `true` if the condition is met, `false` otherwise.
    */
    fun isZeroArray(nums: IntArray, queries: Array<IntArray>): Boolean {
        val n = nums.size
        val diff = IntArray(n + 1)

        for (query in queries) {
            val (l, r) = query
            diff[l] += 1
            if (r + 1 < n) {
                diff[r + 1] -= 1
            }
        }

        var total = 0
        for (i in nums.indices) {
            total += diff[i]
            if (total < nums[i]) {
                return false
            }
        }

        return true
    }
}
