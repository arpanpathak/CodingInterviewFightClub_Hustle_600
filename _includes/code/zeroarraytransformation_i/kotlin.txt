package array.prefixsum

class ZeroArrayTransformation_I {

/**
 * is Zero Array — executes the core logic of this algorithm on the provided input.
 *
 * @param nums the input array of numbers to process
 * @param queries the queries parameter — a array of integers used in the computation
 * @return `true` if the operation succeeds / condition holds, `false` otherwise
 */
    fun isZeroArray(nums: IntArray, queries: Array<IntArray>): Boolean {
        val n = nums.size
        val diff = IntArray(n + 1)

        for (query in queries) {
            val (l, r) = query
            diff[l] += 1
            if (r + 1 < n) {
                diff[r + 1] -= 1
            }
        }

        var total = 0
        for (i in nums.indices) {
            total += diff[i]
            if (total < nums[i]) {
                return false
            }
        }

        return true
    }
}
