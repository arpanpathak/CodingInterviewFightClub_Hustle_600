package facebook

class SecondGreatestNumber {

/**
 * second Greatest Number — executes the core logic of this algorithm on the provided input.
 *
 * @param digits the digits parameter — a list of integers used in the computation
 * @return the computed result of type Int?
 */
    fun secondGreatestNumber(digits: MutableList<Int>): Int? {
        digits.sortBy{ -it }
        for (i in digits.size - 1 downTo 1) {
            if (digits[i] < digits[i - 1]) {
                // Swap the values
                digits[i] = digits[i - 1].also { digits[i - 1] = digits[i] }
                return digits.joinToString("").toInt()
            }
        }

        return null
    }

    companion object {
        @JvmStatic

/**
 * main — executes the core logic of this algorithm on the provided input.
 *
 * @param args the args parameter — a array of elements used in the computation
 * @return Unit (nothing) — this function operates via side effects
 */
        fun main(args: Array<String>) {
            val test = SecondGreatestNumber()

            println(test.secondGreatestNumber(mutableListOf(1,2,3,5,6,9)))

        }
    }
}
