package facebook

class SecondGreatestNumber {

    /**
    * Solves the Second Greatest Number problem.
    * Takes `digits` (mutable list of integers).
    *
    * @param digits The input mutable list of integers.
    * @return The computed integer result.
    */
    fun secondGreatestNumber(digits: MutableList<Int>): Int? {
        digits.sortBy{ -it }
        for (i in digits.size - 1 downTo 1) {
            if (digits[i] < digits[i - 1]) {
                // Swap the values
                digits[i] = digits[i - 1].also { digits[i - 1] = digits[i] }
                return digits.joinToString("").toInt()
            }
        }

        return null
    }

    companion object {
        @JvmStatic
        /**
        * Entry point for the program.
        *
        * @param args The input Array<String>.
        * @return Unit (no return value, modifies state in-place).
        */
        fun main(args: Array<String>) {
            val test = SecondGreatestNumber()

            println(test.secondGreatestNumber(mutableListOf(1,2,3,5,6,9)))

        }
    }
}
