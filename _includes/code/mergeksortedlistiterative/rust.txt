        var interval = 1
        val n = lists.size

                lists[i] = mergeTwoLists(lists[i], lists[i + interval])
            }

        }

    }

        val head = ListNode(0) // Dummy node
        var ptr = head

        var ptr1 = list1
        var ptr2 = list2

        while (ptr1 != None && ptr2 != None) {
            if (ptr1.`val` < ptr2.`val`) {

                ptr1 = ptr1.next
            } else {

        }

        return head.next
    }
