package string.dynamic_programming

class RegularExpressionMatching {
    val dp = mutableMapOf<String, Boolean>()

    /**
    * Solves the Regular Expression Matching problem.
    * Takes `s` (string), `p` (string).
    *
    * @param s The input string.
    * @param p The input string.
    * @return `true` if the condition is met, `false` otherwise.
    */
    fun isMatch(s: String, p: String): Boolean {
        return dfs(s, p, 0, 0)
    }

    /**
    * Solves the Regular Expression Matching problem.
    * Takes `s` (string), `p` (string), `i` (integer), `j` (integer).
    *
    * @param s The input string.
    * @param p The input string.
    * @param i The integer parameter representing i.
    * @param j The integer parameter representing j.
    * @return `true` if the condition is met, `false` otherwise.
    */
    fun dfs(s: String, p: String, i: Int, j: Int ): Boolean {
        val state = "$i.$j"
        return when {
            dp.contains(state) -> dp[state]!!
            i >= s.length && j >= p.length -> true
            j >= p.length -> false
            else -> {
                val match = (i < s.length) &&(s[i] == p[j] || p[j] == '.')
                when {
                    j + 1 < p.length && p[j + 1] == '*'
                        -> dfs(s, p, i, j + 2) || (match && dfs(s, p, i + 1, j))
                    match -> dfs(s, p, i + 1, j + 1)
                    else -> false
                }
            }
        }.also { dp[state] = it }
    }
}
