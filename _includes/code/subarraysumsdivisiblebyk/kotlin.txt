package array.prefixsum


class SubArraySumsDivisibleByK {

/**
 * subarrays Div By K — executes the core logic of this algorithm on the provided input.
 *
 * @param A the first input array for comparison/merging
 * @param K the number of elements/operations to consider (k parameter)
 * @return the computed integer result
 */
    fun subarraysDivByK(A: IntArray, K: Int): Int {
        var ans = 0
        var sum = 0
        val hm = mutableMapOf<Int, Int>()
        hm[0] = 1

        for (num in A) {
            sum += num
            val rem = Math.floorMod(sum, K)
            ans += hm.getOrDefault(rem, 0)
            hm[rem] = hm.getOrDefault(rem, 0) + 1
        }

        return ans
    }
}
