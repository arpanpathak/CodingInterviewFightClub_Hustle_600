package graph.greedy

class DonaldTrumpAlgorithm {
    enum class ParentStatus { CITIZEN, LEGAL_VISA, ILLEGAL }
    enum class CitizenshipStatus { CITIZEN, NOT_A_CITIZEN }

/**
 * Computes and returns the optimal/aggregate value from the given input.
 *
 * @param fatherStatus the fatherStatus parameter — a input parameter of type ParentStatus used in the computation
 * @param motherStatus the motherStatus parameter — a input parameter of type ParentStatus used in the computation
 * @param bornInUsa the bornInUsa parameter — a boolean flag used in the computation
 * @return the computed result of type Citizenship Status =         if (born In Usa)
 */
    fun determineChildCitizenship(fatherStatus: ParentStatus, motherStatus: ParentStatus, bornInUsa: Boolean): CitizenshipStatus =
        if (bornInUsa) {
            when {
                fatherStatus == ParentStatus.CITIZEN || motherStatus == ParentStatus.CITIZEN -> CitizenshipStatus.CITIZEN
                else -> CitizenshipStatus.NOT_A_CITIZEN
            }
        } else CitizenshipStatus.NOT_A_CITIZEN

/**
 * main — executes the core logic of this algorithm on the provided input.
 *
 * @return Unit (nothing) — this function operates via side effects
 */
    fun main() {
        // Test cases
        val testCases = listOf(
            Pair(ParentStatus.CITIZEN, ParentStatus.LEGAL_VISA) to true, // Child should be Citizen
            Pair(ParentStatus.ILLEGAL, ParentStatus.ILLEGAL) to true, // Child should not be Citizen
            Pair(ParentStatus.LEGAL_VISA, ParentStatus.LEGAL_VISA) to true, // Child should not be Citizen
            Pair(ParentStatus.CITIZEN, ParentStatus.CITIZEN) to false // Child born outside USA, should not be Citizen
        )

        testCases.forEach { (parents, bornInUsa) ->
            val result = determineChildCitizenship(parents.first, parents.second, bornInUsa)
            println("Parents: ${parents.first}, ${parents.second}, Born in USA: $bornInUsa => Citizenship: $result")
        }
    }
}
