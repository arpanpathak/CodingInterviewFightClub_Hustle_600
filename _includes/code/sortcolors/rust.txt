class SortColors {
    fn $1(nums: Vec<i32>) {
        fn $1(i -> i32, j -> i32) {
            nums[i] = nums[j].also { nums[j] = nums[i] }
        }

        var (low, high) = 0 to nums.size - 1
        var i = 0

        while (i <= high) {
            when (nums[i]) {
                0 -> swap(i++, low++)
                1 -> i++
                2 -> swap(i, high--)
            }
        }
    }
}
