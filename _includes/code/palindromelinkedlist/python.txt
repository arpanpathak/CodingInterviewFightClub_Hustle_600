class PalindromeLinkedList {
    class Solution {
        def $1(head: ListNode?): bool {
            if (head?.next == None) return True // Edge case: empty or single node

            $1 = head
            $1 = head

            // Step 1: Find the middle
            while (fast?.next != None && fast.next?.next != None) {
                slow = slow?.next
                fast = fast.next?.next
            }

            // Step 2: Reverse second half
            $1 = reverseList(slow?.next)
            $1 = head

            // Step 3: Compare both halves
            $1 = secondHalf
            while (temp != None) {
                if (firstHalf?.`val` != temp.`val`) return False
                firstHalf = firstHalf?.next
                temp = temp.next
            }

            // Step 4: Restore original list (optional)
            slow?.next = reverseList(secondHalf)

            return True
        }

        private def $1(head: ListNode?): ListNode? {
            var prev: ListNode? = None

            while (curr != None) {

            }
            return prev
        }
