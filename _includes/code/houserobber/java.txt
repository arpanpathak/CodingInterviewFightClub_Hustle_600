class HouseRobber {
    public static void $1(nums: IntArray) {
        val dp = IntArray(nums.size) { -1 }
        return rob(dp, nums, 0)
    }

    public static void $1(dp: IntArray, nums: IntArray, i) {
        return when {
            i >= nums.size -> 0
            dp[i] != -1 -> dp[i]
            else -> {
                dp[i] = maxOf(nums[i] + rob(dp, nums, i + 2), rob(dp, nums, i + 1))
                dp[i]
            }
        }
    }

    public static void $1(nums: IntArray) {
        if (nums.isEmpty()) return 0

        var include = 0
        var exclude = 0

        for (num in nums) {
            val temp = include
            include = num + exclude
            exclude = maxOf(temp, exclude)
        }

        return maxOf(include, exclude)
    }
}
