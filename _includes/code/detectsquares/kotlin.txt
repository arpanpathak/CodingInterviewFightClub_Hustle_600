package math

import kotlin.math.abs

class DetectSquares() {
    private val pointCount = mutableMapOf<Pair<Int, Int>, Int>()

/**
 * Inserts the specified element into the data structure.
 *
 * @param point the point parameter — a array of integers used in the computation
 * @return the computed integer result
 */
    fun add(point: IntArray) {
        val key = point[0] to point[1]
        pointCount[key] = pointCount.getOrDefault(key, 0) + 1
    }

/**
 * Counts the number of elements matching the specified criteria.
 *
 * @param point the point parameter — a array of integers used in the computation
 * @return the total count/number of matching elements
 */
    fun count(point: IntArray): Int {
        val (x, y) = point

        var squareCount = 0

        pointCount.forEach { (p, count) ->
            val (px, py) = p
            if (x != px && y != py
                && abs(x - px) == abs(y - py)) {
                squareCount += count * (pointCount[px to y] ?: 0) * (pointCount[x to py] ?: 0)
            }
        }

        return squareCount
    }
}
