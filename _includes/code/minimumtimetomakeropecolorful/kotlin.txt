package greedy

class MinimumTimeToMakeRopeColorful {
    /**
    * Solves the Minimum Time To Make Rope Colorful problem.
    * Takes `colors` (string), `neededTime` (array of integers).
    *
    * @param colors The input string.
    * @param neededTime The input array of integers.
    * @return The computed integer result.
    */
    /**
    * Solves the Minimum Time To Make Rope Colorful problem.
    * Takes `colors` (string), `neededTime` (array of integers).
    *
    * @param colors The input string.
    * @param neededTime The input array of integers.
    * @return The computed integer result.
    */
    /**
    * Solves the Minimum Time To Make Rope Colorful problem.
    * Takes `colors` (string), `neededTime` (array of integers).
    *
    * @param colors The input string.
    * @param neededTime The input array of integers.
    * @return The computed integer result.
    */
    /**
    * Solves the Minimum Time To Make Rope Colorful problem.
    * Takes `colors` (string), `neededTime` (array of integers).
    *
    * @param colors The input string.
    * @param neededTime The input array of integers.
    * @return The computed integer result.
    */
    fun minCost(colors: String, neededTime: IntArray): Int {
        var minTime = 0
        for (i in 1 until neededTime.size) {
            if (colors[i] == colors[i - 1]) {
                minTime += minOf(neededTime[i], neededTime[i - 1])
                neededTime[i] = maxOf(neededTime[i], neededTime[i - 1])
            }
        }

        return minTime
    }
}
